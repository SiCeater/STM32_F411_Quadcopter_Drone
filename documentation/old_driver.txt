#include "imu.h"

uint8_t shtpData[MAX_PACKET_SIZE]; // shtpData array, BNO085 can respond with upto 270 bytes, avoid overflow
int16_t receivedDataLength;
uint8_t shtpHeader [4];

void IMU_SPIBus_init(void)
{
	LL_SPI_Enable(BNO085_SPI_CHANNEL); // on active le SPI
	CHIP_DESELECT(BNO085); // on force la desactivation de la communication
	RESET_HIGH(); // on desactive le reset
}

uint8_t BNO085_SPI_bus_read_write_8(uint8_t data)
{
	while(LL_SPI_IsActiveFlag_TXE(BNO085_SPI_CHANNEL)==RESET);
	LL_SPI_TransmitData8(BNO085_SPI_CHANNEL, data);
	
	while(LL_SPI_IsActiveFlag_RXNE(BNO085_SPI_CHANNEL)==RESET);
	return LL_SPI_ReceiveData8(BNO085_SPI_CHANNEL);
}

//Blocking wait for BNO085 to assert (pull low) the INT pin
//indicating it's ready for comm. Can take more than 104ms after a hardware reset
uint8_t BNO085_wait_response(void)
{
	char buffer_debug[100];
	for (uint32_t counter = 0; counter < 50; counter++) //Don't got more than 255
	{
		if (LL_GPIO_IsInputPinSet(BNO085_INT_PORT, BNO085_INT_PIN) == 0)
		{
			if(debug>=2)
				print_to_console("\nData available\n",16);
			return (1);
		}
		if (debug>=3)
		{
			sprintf(buffer_debug,"\nSPI Wait %ld\n", counter);
			print_to_console(buffer_debug,strlen(buffer_debug));
		}
	}
	if(debug>=2)
		print_to_console("\nData not available\n",20);
	return (0);
}

uint16_t BNO085_read_response(void)
{
    uint8_t incoming;
	char buffer_debug[100];
    for (u_int16_t i = 0; i < MAX_PACKET_SIZE; i++) // on nettoie le buffer de reception
        shtpData[i]=0;
	if (LL_GPIO_IsInputPinSet(BNO085_INT_PORT, BNO085_INT_PIN) == 1)
    {
        // un truc chiant s'est passé imprime un truc sur la console neuille
        return (0); //Data is not available
    }
	CHIP_SELECT(BNO085);
	//Get the first four bytes, aka the packet header
	uint8_t packetLSB = BNO085_SPI_bus_read_write_8(0);
	uint8_t packetMSB = BNO085_SPI_bus_read_write_8(0);
	uint8_t channelNumber = BNO085_SPI_bus_read_write_8(0);
	uint8_t sequenceNumber = BNO085_SPI_bus_read_write_8(0); //Not sure if we need to store this or not
    shtpHeader[0]=packetLSB;
    shtpHeader[1]=packetMSB;
    shtpHeader[2]=channelNumber;
    shtpHeader[3]=sequenceNumber;
	//Calculate the number of data bytes in this packet
	receivedDataLength = ((uint16_t)packetMSB << 8 | packetLSB);
	receivedDataLength &= 0x7fff; //Clear the MSbit.
	//This bit indicates if this package is a continuation of the last. Ignore it for now.
	//TODO catch this as an error and exit
	if (receivedDataLength == 0)
	{
		//Packet is empty reponse vide imprime un truc sur la console la aussi neuille
		return (0); //All done
	}
    receivedDataLength -= 4; //Remove the header bytes from the data count
	if (debug>=2)
	{
		sprintf(buffer_debug,"length: %d\n", receivedDataLength);
		print_to_console(buffer_debug,strlen(buffer_debug));
        sprintf(buffer_debug,"%d %d %d %d ",packetLSB,packetMSB,channelNumber,sequenceNumber );
		print_to_console(buffer_debug,strlen(buffer_debug));
	}

	for (uint16_t dataSpot = 0; dataSpot < receivedDataLength; dataSpot++) //Read incoming data into the shtpData array
	{
		incoming = BNO085_SPI_bus_read_write_8(0xFF);
		if (debug>=3)
		{
			sprintf(buffer_debug,"%d ", incoming);
			print_to_console(buffer_debug,strlen(buffer_debug));
		}
		if (dataSpot < MAX_PACKET_SIZE)	//BNO085 can respond with upto 270 bytes, avoid overflow
			shtpData[dataSpot] = incoming; //Store data into the shtpData array
	}
	if(debug>=2)
		print_to_console("\n",1);
	CHIP_DESELECT(BNO085); //Release BNO085
	return receivedDataLength; //We're done!
}

uint8_t BNO085_check() 
{
    uint8_t whoami = 0;
    //on select, on transmet, on deselect on attend int, on reselect on lis et on deselect
    // Lecture du registre WHO_AM_I
    CHIP_SELECT(BNO085);
    while (!LL_SPI_IsActiveFlag_TXE(BNO085_SPI_CHANNEL));
    LL_SPI_TransmitData8(BNO085_SPI_CHANNEL,0xF9); // Commande spécifique au WHO_AM_I
    CHIP_DESELECT(BNO085);
    BNO085_wait_response();
    CHIP_SELECT(BNO085);
    while (!LL_SPI_IsActiveFlag_RXNE(BNO085_SPI_CHANNEL));
    whoami = LL_SPI_ReceiveData8(BNO085_SPI_CHANNEL); // Lecture de la réponse
    CHIP_DESELECT(BNO085);
    return whoami;
}

void BNO085_Set_Feature(uint8_t feature_id, uint32_t interval_us, uint8_t specific_config)
{
    uint8_t cmd[20] = 
    {
        0x14, 0x00,              // Longueur totale (20 octets)
        0x02,                    // Canal (Input/Output)
        0x00,                    // Numéro de séquence (incrémenté si nécessaire)
        0xFD,                    // Command ID (Set Feature Command)
        feature_id, 0x00,        // Feature Report ID
        (uint8_t)(interval_us & 0xFF),
        (uint8_t)((interval_us >> 8) & 0xFF),
        (uint8_t)((interval_us >> 16) & 0xFF),
        (uint8_t)((interval_us >> 24) & 0xFF), // Intervalle en microsecondes
        specific_config, 0x00, 0x00, 0x00,    // Paramètres spécifiques (par défaut ici)
        0x00, 0x00, 0x00, 0x00                // Remplissage
    };

    // Vérification si le BNO085 est prêt
    if (!BNO085_wait_response()) {
        print_to_console("⚠️ BNO085 pas prêt, annulation de Set Feature\n", 45);
        return;
    }

    // Envoi de la commande
    CHIP_SELECT(BNO085);
    for (uint8_t i = 0; i < sizeof(cmd); i++) {
        BNO085_SPI_bus_read_write_8(cmd[i]);
    }
    CHIP_DESELECT(BNO085);

    // Attendre une réponse du BNO085 pour confirmer la prise en compte
    if (!BNO085_wait_response()) {
        print_to_console("⚠️ Set Feature: Pas de réponse du BNO085\n", 42);
        return;
    }

    // Lire la réponse pour voir si le capteur a bien accepté la commande
    if (BNO085_read_response() == 0) {
        print_to_console("⚠️ Set Feature: Réponse vide\n", 28);
        return;
    }

    print_to_console("✅ Capteur configuré avec succès\n", 35);
}

// Fonction pour envoyer une requête et récupérer les capteurs activés
void BNO085_Get_Active_Sensors() 
{
    if (!BNO085_wait_response()) {
        print_to_console("BNO085 pas prêt, annulation de la requête.\n", 44);
        return;
    }

    uint8_t shtp_command[] = { SHTP_REPORT_GET_FEATURE, 0x00 };

    CHIP_SELECT(BNO085);
    for (uint8_t i = 0; i < sizeof(shtp_command); i++) {
        BNO085_SPI_bus_read_write_8(shtp_command[i]);
    }
    CHIP_DESELECT(BNO085);

    if (!BNO085_wait_response()) {
        print_to_console("⚠️ Pas de réponse après la requête.\n", 34);
        return;
    }

    if (BNO085_read_response() == 0) {
        print_to_console("⚠️ Réponse vide, capteur inactif ?\n", 32);
        return;
    }

    print_to_console("📡 Capteurs activés et leur fréquence :\n", 37);
    char buffer[100];

    for (int i = 0; i < receivedDataLength; i += 16) 
    {
        uint8_t sensor_id = shtpData[i + 1];
        uint32_t report_interval = 
            (shtpData[i + 5] | (shtpData[i + 6] << 8) | 
             (shtpData[i + 7] << 16) | (shtpData[i + 8] << 24));

        if (sensor_id == 0xFF) break;

        snprintf(buffer, sizeof(buffer), "Capteur ID: 0x%02X - Fréquence: %.2f Hz\n", 
                 sensor_id, 1000000.0 / report_interval);
        print_to_console(buffer, strlen(buffer));
    }
}
